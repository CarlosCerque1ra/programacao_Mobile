import flet as ft

def main(page: ft.Page):
    page.title = "Campo de Texto"
    page.padding = ft.padding.only(top=40, left=20, right=20, bottom=20)  # Padding superior para √°rea segura

    # Criando um campo onde o usu√°rio pode digitar
    campo_nome = ft.TextField(
        label="Digite seu nome aqui",  # Texto de orienta√ß√£o
        width=300,  # Largura do campo
        border_color=ft.Colors.BLUE  # Cor da borda
    )

    # Texto que mostrar√° a resposta
    resposta = ft.Text(
        value="",  # Inicialmente vazio
        size=18,
        text_align=ft.TextAlign.CENTER
    )

    def processar_nome(evento):
        """
        Fun√ß√£o que pega o texto digitado pelo usu√°rio e faz algo com ele.
        """
        # Pegando o valor digitado no campo
        nome_digitado = campo_nome.value

        # Verificando se o usu√°rio realmente digitou algo
        if nome_digitado == "" or nome_digitado is None:
            resposta.value = "‚ö†Ô∏è Por favor, digite seu nome!"
            resposta.color = ft.Colors.RED
        elif len(nome_digitado) < 2:
            resposta.value = "‚ö†Ô∏è Nome muito curto!"
            resposta.color = ft.Colors.ORANGE
        else:
            resposta.value = f"‚úÖ Ol√°, {nome_digitado}! Prazer em conhec√™-lo(a)!"
            resposta.color = ft.Colors.GREEN

        # Atualizando a interface
        page.update()

    # Bot√£o para processar o nome
    botao_ok = ft.ElevatedButton(
        text="Confirmar",
        on_click=processar_nome,
        width=150
    )

    # Adicionando elementos √† p√°gina
    page.add(
        ft.Text("Vamos nos conhecer! üòä", size=22, weight=ft.FontWeight.BOLD),
        campo_nome,
        botao_ok,
        resposta
    )

ft.app(target=main)
```

import flet as ft

def main(page: ft.Page):
	page.title = "Seletor de Cores"
	page.padding = 20

	# Container que mudar√° de cor (como uma caixa colorida)
	caixa_colorida = ft.Container(
		content=ft.Text(
			"Escolha uma cor! üé®",
			color=ft.Colors.WHITE,
			size=18,
			text_align=ft.TextAlign.CENTER
		),
		bgcolor=ft.Colors.GREY,  # Cor inicial
		width=300,
		height=100,
		border_radius=15,  # Bordas arredondadas
		alignment=ft.alignment.center  # Centralizar o texto dentro da caixa
	)

	def cor_selecionada(evento):
		"""
		Esta fun√ß√£o √© executada sempre que o usu√°rio escolhe uma cor.
		"""
		# Pegando qual cor foi selecionada
		cor_escolhida = evento.control.value

		# Dicion√°rio com as cores dispon√≠veis
		# √â como uma "lista de correspond√™ncia" entre nome e cor real
		cores_disponiveis = {
			"Azul": ft.Colors.BLUE,
			"Verde": ft.Colors.GREEN,
			"Vermelho": ft.Colors.RED,
			"Roxo": ft.Colors.PURPLE,
			"Laranja": ft.Colors.ORANGE,
			"Rosa": ft.Colors.PINK
		}

		# Mudando a cor da caixa
		caixa_colorida.bgcolor = cores_disponiveis[cor_escolhida]
		caixa_colorida.content.value = f"Cor selecionada: {cor_escolhida} üé®"

		page.update()

	# Criando a lista suspensa (dropdown)
	seletor_cor = ft.Dropdown(
		label="Escolha uma cor:",
		width=200,
		options=[
			ft.dropdown.Option("Azul"),
			ft.dropdown.Option("Verde"),
			ft.dropdown.Option("Vermelho"),
			ft.dropdown.Option("Roxo"),
			ft.dropdown.Option("Laranja"),
			ft.dropdown.Option("Rosa")
		],
		on_change=cor_selecionada  # Fun√ß√£o que ser√° executada quando escolher
	)

	# Adicionando elementos √† p√°gina
	page.add(
		ft.Text("Seletor de Cores M√°gico! üé®", size=24, weight=ft.FontWeight.BOLD),
		seletor_cor,
		caixa_colorida
	)

ft.app(target=main)

```

import flet as ft

		"Organizando Elementos na Tela üñ•Ô∏è",
		size=24,
		weight=ft.FontWeight.BOLD,

		spacing=20  # Espa√ßo entre os bot√µes
	)

	# Criando algumas caixas coloridas em coluna

		content=ft.Text("Caixa 1", color=ft.Colors.WHITE),

import flet as ft

			ft.dropdown.Option("Lexus üöô"),
			ft.dropdown.Option("Ford üöò"),
			ft.dropdown.Option("Chevrolet üöï"),
			ft.dropdown.Option("Nissan üöñ"),
		]
	)

		content=ft.Text("Preencha os dados acima"),
		bgcolor=ft.Colors.GREY_100,
		padding=33,
		border_radius=15,
		width=350,
		visible=False
	)

	def criar_perfil(evento):
		"""Valida dados e cria o perfil visual"""
		if not campo_nome.value or len(campo_nome.value) < 2:
			mostrar_erro("Nome deve ter pelo menos 2 caracteres")
			return
		if not campo_idade.value:
			mostrar_erro("Idade √© obrigat√≥ria")
			return
		try:
			idade = int(campo_idade.value)
			if idade < 1 or idade > 120:
				mostrar_erro("Idade deve estar entre 1 e 120 anos")
				return
		except ValueError:
			mostrar_erro("Idade deve ser um n√∫mero")
			return
		if not dropdown_hobby.value:
			mostrar_erro("Selecione uma marca de carro favorita")
			return

	def mostrar_erro(mensagem):
		"""Mostra mensagem de erro"""
		cartao_perfil.content = ft.Column([
			ft.Icon(ft.Icons.ERROR, color=ft.Colors.RED, size=40),
			ft.Text(f"‚ö†Ô∏è {mensagem}", color=ft.Colors.RED, text_align=ft.TextAlign.CENTER)
		], horizontal_alignment=ft.CrossAxisAlignment.CENTER)

		"""Cria o cart√£o visual do perfil"""

		elif idade < 66:

		else:

			cor_icone = ft.Colors.PURPLE

			ft.Icon(ft.Icons.PERSON, size=60, color=cor_icone),
			ft.Text(campo_nome.value, size=20, weight=ft.FontWeight.BOLD),
			ft.Text(f"{idade} anos - {categoria}", size=14, color=ft.Colors.GREY_600),
			ft.Text(f"Marca: {dropdown_hobby.value}", size=14),
			ft.Container(
				content=ft.Text("Perfil criado! ‚ú®", color=ft.Colors.WHITE),
				bgcolor=cor_icone,
				padding=10,
				border_radius=20
			)
		], horizontal_alignment=ft.CrossAxisAlignment.CENTER, spacing=10)
		cartao_perfil.bgcolor = ft.Colors.WHITE
		cartao_perfil.visible = True
		page.update()

		"""Limpa todos os campos"""

		campo_idade.value = ""
		dropdown_hobby.value = None
		cartao_perfil.visible = False
		cartao_perfil.value = None
		page.update()

	linha_botoes = ft.Row([
		ft.ElevatedButton(
			"‚ú® Criar Perfil",

			bgcolor=ft.Colors.BLUE,
			color=ft.Colors.WHITE,
			width=140
		),
		ft.ElevatedButton(
			"üßπ Limpar",
			on_click=limpar_campos,
			bgcolor=ft.Colors.GREY,
			color=ft.Colors.WHITE,
			width=140
		)
	], alignment=ft.MainAxisAlignment.CENTER, spacing=20)

	# Layout principal
	layout_principal = ft.Column([
		ft.Text("üë§ Criador de Perfil", size=26, weight=ft.FontWeight.BOLD),
		ft.Text("Preencha seus dados para criar seu perfil personalizado!",
				size=14, color=ft.Colors.GREY_600, text_align=ft.TextAlign.CENTER),
		ft.Container(height=20),
		campo_nome,
		campo_idade,
